• 동일한 값으로 구성된 배열 두개더라도 서로 다른 배열이다(Imutable) 
              //a와 b는 다른 배열이다
          let a=[1,2,3]
          let b=[1,2,3]
  • 배열은 참조형이다 
              //a와 b는 같은배열
          let a = [1,2,3]
          let b = a
  • 배열 선언방법 
              let a = []
          let a = new Array(3)
  • 배열의 인덱스는 0부터 시작, 갯수 -1 으로 끝 
              //let arr = [1,2,3]
          arr[0] = 1
          arr[2] = 3
  • 배열의 값을 추가, 제거 
              arr.push() //배열 뒤 추가 
          arr.unshift() // 배열 앞 추가
          arr.pop() // 배열 뒤 제거
          arr.shift() // 배열 앞 제거
  • splice(배열의 값을 추가, 제거 혹은 변경) 
              arr = [1,2,3]
          arr.splice(1,2) //1번쨰부터 2개 지움
          => arr = [1]
          arr.splice(2,1,4) //2번쨰부터 1개를 지우고 4 추가
           => arr = [1,2,4]
  • split , 글자를 이용해 배열생성 <== 글자를 쪼개서 배열로 만든다. 
              let str = 123456
          let arr = str.split('')
          arr = [1,2,3,4,5,6]
  • slice, 배열과 문자열의 특정 범위를 반환(return)한다. 
              let arr = ['가', '나', '다', '라', '마'];
          let arr2 = arr.slice(1, 3);//arr배열의 1번째부터 2번까지 반환
          console.log(arr2); // [가,나,다]
          let str = '가나다라마바사';
          let str2 = str.slice(0, -2);//문자의 0번쩨부터 뒤에서 3번전째까지 반환
          console.log(str2);가나다라마
  • join, 배열안의 값들을 문자형으로 반환 
                let arr = ['가', '나', '다'];
            let str = arr.join(); 
            console.log(str)//가,나,다
            let str2 = arr.join(''); 
            console.log(str2)//가나다
            let str3 = arr.join('-');
            console.log(str3)//가-나-다
  • indexOf, 배열의 값과 일치하는 첫번째 인덱스를 반환 
    •             let arr = ['가', '나', '다', '가', '나', '다'];
            let idx = arr.indexOf('나');
            console.log(idx); //1
            idx = arr.indexOf('라');
            console.log(idx); //-1 <--없다는 뜻
